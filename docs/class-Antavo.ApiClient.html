Posts an event to the Antavo Loyalty Events API.

    'id' => '8eb1b522',
    'first_name' => 'John',
    'last_name' => 'Doe',
    'email' => 'john.doe@example.com'
);
$client = new Antavo\ApiClient('API_KEY', 'API_SECRET');
try {
    $result = $client->postEvent('opt_in', $customer['id'], array(
        'first_name' => $customer['first_name'],
